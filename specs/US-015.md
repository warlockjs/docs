### 2. Configuration

```typescript
// src/config/database.ts
export default {
  driver: env("DB_DRIVER", "mongodb"), // or "postgresql"

  // MongoDB
  host: env("DB_HOST", "127.0.0.1"),
  port: env("DB_PORT", 27017),
  name: env("DB_NAME", "myapp"),

  // PostgreSQL
  // host: env("PG_HOST"),
  // port: env("PG_PORT", 5432),
  // database: env("PG_NAME"),
  // user: env("PG_USER"),
  // password: env("PG_PASSWORD"),
};
```

### 3. Models

```typescript
// src/app/posts/models/post/post.model.ts
import { Model } from "@warlock.js/cascade";

export class Post extends Model {
  static collection = "posts"; // MongoDB
  // static table = "posts";    // PostgreSQL

  static schema = {
    title: String,
    content: String,
    authorId: "objectId", // or "integer" for PG
  };
}
```

### 4. Decorators

```typescript
import { Timestamps, SoftDeletes } from "@warlock.js/cascade";

@Timestamps()
@SoftDeletes()
export class Post extends Model {
  // ...
}
```
