## Code Example Pattern

Routes file:

```typescript
import { router } from "@warlock.js/core";
import { createPostController } from "./controllers/create-post.controller";
import { listPostsController } from "./controllers/list-posts.controller";

router.get("/posts", listPostsController);
router.post("/posts", createPostController);
```

Controller file:

```typescript
import { type RequestHandler } from "@warlock.js/core";
import { v } from "@warlock.js/seal";
import { Post } from "../models/post";

export const createPostController: RequestHandler = async (
  request,
  response
) => {
  const post = await Post.create(request.validated());
  return response.success({ post });
};

createPostController.validation = {
  schema: v.object({
    title: v.string().required(),
    content: v.string().required(),
  }),
};
```
