### 2. Configuration

```typescript
// src/config/auth.ts
export default {
  jwt: {
    secret: env("JWT_SECRET"),
    expiresIn: "7d",
  },

  userModel: User,

  // User types
  userTypes: ["user", "admin"],
};
```

### 3. Guards (Middleware)

```typescript
import { authMiddleware } from "@warlock.js/auth";
import { router } from "@warlock.js/core";

// Protect routes
router.group(
  {
    middleware: [authMiddleware()], // Any authenticated user
  },
  () => {
    router.get("/profile", profileController);
  }
);

// Specific user type
router.group(
  {
    middleware: [authMiddleware("admin")],
  },
  () => {
    router.get("/admin/dashboard", dashboardController);
  }
);
```

### 4. Roles & Permissions
