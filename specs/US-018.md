### 2. Configuration

```typescript
// src/config/cache.ts
export default {
  driver: "redis",

  redis: {
    host: env("REDIS_HOST", "127.0.0.1"),
    port: env("REDIS_PORT", 6379),
    password: env("REDIS_PASSWORD"),
  },

  prefix: "myapp:",
  ttl: 3600, // Default TTL in seconds
};
```

### 3. Usage

```typescript
import { cache } from "@warlock.js/core";

// Set cache
await cache.set("key", value, 3600); // 1 hour

// Get cache
const value = await cache.get("key");

// Delete
await cache.forget("key");

// Check exists
const exists = await cache.has("key");

// Remember (get or set)
const data = await cache.remember("key", 3600, async () => {
  return await fetchExpensiveData();
});
```

### Tagged Cache

```typescript
// Tag cache entries
await cache.tags(["users", "posts"]).set("key", value);

// Flush by tag
await cache.tags(["users"]).flush();
```
