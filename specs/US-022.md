### 2. Setup

```typescript
// src/config/communicator.ts
export default {
  driver: "rabbitmq",

  rabbitmq: {
    url: env("RABBITMQ_URL"),
    // or
    host: env("RABBITMQ_HOST"),
    port: env("RABBITMQ_PORT"),
    user: env("RABBITMQ_USER"),
    password: env("RABBITMQ_PASSWORD"),
  },
};
```

### 3. Publishing

```typescript
import { herald } from "@warlock.js/herald";

// Define message type
interface UserCreatedMessage {
  userId: string;
  email: string;
}

// Publish
await herald.publish<UserCreatedMessage>("user.created", {
  userId: "123",
  email: "user@example.com",
});
```

### 4. Subscribing

```typescript
import { herald } from "@warlock.js/herald";

// Subscribe to channel
herald.subscribe("user.created", async (message: UserCreatedMessage) => {
  console.log("User created:", message.userId);
  // Send welcome email, etc.
});
```
