### 2. Configuration

```typescript
  driver: "postgresql",

  port: env("PG_PORT", 5432),
  database: env("PG_NAME"),
  user: env("PG_USER"),
  password: env("PG_PASSWORD"),
};
```

### 3. Models

```typescript
import { Model } from "@warlock.js/cascade";

export class Post extends Model {
  static table = "posts"; // table instead of collection

  static schema = {
    id: "serial", // Auto-increment ID
    title: "string",
    content: "text",
    authorId: "integer", // Foreign key
    createdAt: "timestamp",
  };
}
```

**Key differences:**

- `table` instead of `collection`
- `serial` for auto-increment IDs
- `integer` instead of `objectId` for relations
- SQL types: `text`, `timestamp`, `boolean`, etc.

### 4. Migrations

```typescript
// migrations/2024_01_01_create_posts.ts
import { Migration } from "@warlock.js/cascade";

export default class CreatePosts extends Migration {
  async up() {
    await this.createTable("posts", (table) => {
      table.serial("id").primary();
      table.string("title");
      table.text("content");
      table.integer("author_id").references("users.id");
      table.timestamps();
    });
  }

  async down() {
    await this.dropTable("posts");
  }
}
```

### 5. Queries

```typescript
// Most queries work the same
const posts = await Post.where("status", "published").get();

// PostgreSQL-specific
const posts = await Post.whereRaw(
  "created_at > NOW() - INTERVAL '7 days'"
).get();
```
