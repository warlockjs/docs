### 2. Configuration

```typescript
// src/config/mail.ts
export default {
  host: env("MAIL_HOST"),
  port: env("MAIL_PORT", 465),
  secure: env("MAIL_SECURE", true),
  auth: {
    user: env("MAIL_USERNAME"),
    pass: env("MAIL_PASSWORD"),
  },
  from: {
    name: env("MAIL_FROM_NAME"),
    address: env("MAIL_FROM_ADDRESS"),
  },
};
```

### 3. Sending Emails

```typescript
import { mail } from "@warlock.js/core";

await mail.send({

});
```

### 4. React Templates (NEW)

```tsx
// src/app/users/mail/WelcomeMail.tsx
import React from "react";

interface WelcomeMailProps {
  name: string;
  verificationUrl: string;
}

export function WelcomeMail({ name, verificationUrl }: WelcomeMailProps) {
  return (
    <div>
      <h1>Welcome, {name}!</h1>
      <p>Click below to verify your email:</p>
      <a href={verificationUrl}>Verify Email</a>
    </div>
  );
}
```

```typescript
// Sending React email
import { mail } from "@warlock.js/core";
import { WelcomeMail } from "./mail/WelcomeMail";
```
